Advancement Creation # Load events are the most appropriate events to use. Loading the advancement tab should not affect the server much.
on load:
 create advancement tab named "my_tab":
  create advancement named "root":
   set root of advancement to true
   set background of advancement to gold block
   set title of advancement to "&dMy new advancement! :)"
   set description of advancement to "&7This is a pretty cool advancement, huh?"
   build advancement
  create advancement named "base1":
   set icon of advancement to green wool
   set title of advancement to "&bBase 1"
   # Treating the description as a list creates separate lines.
   set description of advancement to "&aThis is a base advancement" and "&amade with &2skript-advancements&a!"
   set column of advancement to 1
   # You must specify the parent of a base advancement for it to register.
   set parent of advancement to "my_tab/root"
   build advancement
  create advancement named "base2":
   # Items with NBTs will appear with their set NBTs as an icon.
   set icon of advancement to dirt of mending
   set title of advancement to "&bBase 2"
   set description of advancement to "&cAnother base advancement!"
   set column of advancement to 1
   set row of advancement to 1
   set parent of advancement to "my_tab/root"
   build advancement
  create advancement named "base2":
   # Items with NBTs will appear with their set NBTs as an icon.
   set icon of advancement to dirt of mending
   set title of advancement to "&bBase 3"
   set description of advancement to "&4Very cool multi parent advancement :)"
   set column of advancement to 2
   set row of advancement to 0.5
   # Advancements can have multiple parent advancements. These advancements cannot be a root advancement!
   set parents of advancement to "my_tab/base1" and "my_tab/base2"
   build advancement
  build tab
